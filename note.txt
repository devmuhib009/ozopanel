Get post objects
// Replace 'your_post_type_name' with the name of the post type you want to retrieve the label for
$post_type_name = 'your_post_type_name';

// Get the post type object based on the post type name
$post_type_object = get_post_type_object($post_type_name);

// Check if the post type object exists
if ($post_type_object) {
    // Output the post type label
    echo 'Post Type Label: ' . $post_type_object->labels->name;
} else {
    // Output a message if the post type does not exist
    echo 'Invalid Post Type Name';
}
